local function GetCharacter()
    return LocalPlayer.Character
end

local function GetHumanoid()
    local c = GetCharacter()
    if c then return c:FindFirstChildOfClass("Humanoid") end
    return nil
end

local function GetRootPart()
    local c = GetCharacter()
    if c then return c:FindFirstChild("HumanoidRootPart") end
    return nil
end

local function IsAlive()
    local h = GetHumanoid()
    if h and h.Health > 0 then return true end
    return false
end

local function GetNearestPlayer(maxDist)
    local root = GetRootPart()
    if not root then return nil, math.huge end
    local best = nil
    local bestDist = maxDist or math.huge
    for _, p in ipairs(Players:GetPlayers()) do
        if p ~= LocalPlayer and p.Character then
            local r = p.Character:FindFirstChild("HumanoidRootPart")
            local h = p.Character:FindFirstChildOfClass("Humanoid")
            if r and h and h.Health > 0 then
                local d = (root.Position - r.Position).Magnitude
                if d < bestDist then
                    best = p
                    bestDist = d
                end
            end
        end
    end
    return best, bestDist
end

local function FindPlayerByName(name)
    name = name:lower()
    for _, p in ipairs(Players:GetPlayers()) do
        if p ~= LocalPlayer then
            if p.Name:lower():find(name) or p.DisplayName:lower():find(name) then
                return p
            end
        end
    end
    return nil
end

local function SimulateClick()
    pcall(function()
        local VIM = game:GetService("VirtualInputManager")
        VIM:SendMouseButtonEvent(Mouse.X, Mouse.Y, 0, true, game, 0)
        task.delay(0.03, function()
            pcall(function()
                VIM:SendMouseButtonEvent(Mouse.X, Mouse.Y, 0, false, game, 0)
            end)
        end)
    end)
end

local function SimulateKey(keyCode, duration)
    duration = duration or 0.08
    pcall(function()
        local VIM = game:GetService("VirtualInputManager")
        VIM:SendKeyEvent(true, keyCode, false, game)
        task.delay(duration, function()
            pcall(function()
                VIM:SendKeyEvent(false, keyCode, false, game)
            end)
        end)
    end)
end

local function FindBall()
    for _, obj in ipairs(workspace:GetDescendants()) do
        if obj:IsA("BasePart") then
            local n = obj.Name:lower()
            if n == "ball" or n == "bladeball" or n == "blade" then
                return obj
            end
        end
    end
    for _, folder in ipairs(workspace:GetChildren()) do
        if folder:IsA("Folder") or folder:IsA("Model") then
            local fn = folder.Name:lower()
            if fn:find("ball") or fn:find("blade") or fn:find("projectile") then
                for _, child in ipairs(folder:GetDescendants()) do
                    if child:IsA("BasePart") then
                        return child
                    end
                end
            end
        end
    end
    return nil
end

local function GetBallSpeed(ball)
    if not ball or not ball.Parent then return 0 end
    return ball.AssemblyLinearVelocity.Magnitude
end

local function IsBallComingToMe(ball)
    if not ball or not ball.Parent then return false end
    local myRoot = GetRootPart()
    if not myRoot then return false end
    local dist = (ball.Position - myRoot.Position).Magnitude
    if dist > Config.ParryDistance then return false end
    local ballVel = ball.AssemblyLinearVelocity
    if ballVel.Magnitude < 5 then return false end
    local toMe = (myRoot.Position - ball.Position).Unit
    local ballDir = ballVel.Unit
    local dot = toMe:Dot(ballDir)
    if dot > 0.3 then return true end
    return false
end

local function GetBallETA(ball)
    if not ball or not ball.Parent then return math.huge end
    local myRoot = GetRootPart()
    if not myRoot then return math.huge end
    local dist = (ball.Position - myRoot.Position).Magnitude
    local speed = ball.AssemblyLinearVelocity.Magnitude
    if speed < 1 then return math.huge end
    return dist / speed
end

local function GetBallDistToMe(ball)
    if not ball or not ball.Parent then return math.huge end
    local myRoot = GetRootPart()
    if not myRoot then return math.huge end
    return (ball.Position - myRoot.Position).Magnitude
end

local function FindParryRemote()
    local function Search(parent)
        if not parent then return end
        pcall(function()
            for _, child in ipairs(parent:GetDescendants()) do
                if child:IsA("RemoteEvent") then
                    local n = child.Name:lower()
                    if n:find("parry") or n:find("deflect") or n:find("block") or n:find("hit") or n:find("swing") or n:find("click") then
                        if not ParryRemote then
                            ParryRemote = child
                        end
                    end
                end
            end
        end)
    end
    Search(ReplicatedStorage)
    pcall(function() Search(LocalPlayer.PlayerScripts) end)
    pcall(function()
        local char = GetCharacter()
        if char then
            for _, tool in ipairs(char:GetChildren()) do
                if tool:IsA("Tool") then Search(tool) end
            end
        end
        local bp = LocalPlayer:FindFirstChild("Backpack")
        if bp then
            for _, tool in ipairs(bp:GetChildren()) do
                if tool:IsA("Tool") then Search(tool) end
            end
        end
    end)
end

task.spawn(function()
    task.wait(3)
    FindParryRemote()
end)

local function FireParry()
    if ParryRemote then
        pcall(function() ParryRemote:FireServer() end)
    end
    SimulateClick()
    pcall(function()
        local char = GetCharacter()
        if char then
            for _, tool in ipairs(char:GetChildren()) do
                if tool:IsA("Tool") then tool:Activate() end
            end
        end
    end)
end

local function RunAutoParry()
    if ParryDebounce or not IsAlive() then return end
    local ball = FindBall()
    if not ball then return end
    if IsBallComingToMe(ball) then
        local eta = GetBallETA(ball)
        if eta < Config.ParryTiming then
            ParryDebounce = true
            FireParry()
            task.delay(0.4, function() ParryDebounce = false end)
        end
    end
end

local function RunSmartParry()
    if ParryDebounce or not IsAlive() then return end
    local ball = FindBall()
    if not ball then return end
    if not IsBallComingToMe(ball) then return end

    local dist = GetBallDistToMe(ball)
    local speed = GetBallSpeed(ball)
    local eta = GetBallETA(ball)

    -- Dynamic timing based on ball speed
    local dynamicTiming = Config.ParryTiming
    if speed > 150 then
        dynamicTiming = dynamicTiming * 0.6
    elseif speed > 100 then
        dynamicTiming = dynamicTiming * 0.8
    end

    -- Also check raw distance as fallback
    local distThreshold = math.max(8, speed * 0.06)

    if eta < dynamicTiming or dist < distThreshold then
        ParryDebounce = true
        FireParry()
        task.delay(0.35, function() ParryDebounce = false end)
    end
end

local function RunAutoDodge()
    if DodgeDebounce or not IsAlive() then return end
    local ball = FindBall()
    if not ball then return end

    local myRoot = GetRootPart()
    if not myRoot then return end

    local dist = GetBallDistToMe(ball)
    if dist > Config.DodgeDistance then return end

    if IsBallComingToMe(ball) then
        DodgeDebounce = true

        -- Move perpendicular to ball direction
        local ballDir = ball.AssemblyLinearVelocity.Unit
        local right = Vector3.new(-ballDir.Z, 0, ballDir.X)

        -- Pick random side
        if math.random() > 0.5 then
            right = -right
        end

        pcall(function()
            myRoot.CFrame = myRoot.CFrame + (right * 8)
        end)

        task.delay(0.5, function() DodgeDebounce = false end)
    end
end

local function RunAutoSpam()
    if not IsAlive() then return end
    local now = tick()
    if now - lastSpamTick < Config.SpamDelay then return end
    lastSpamTick = now
    SimulateClick()
    pcall(function()
        local char = GetCharacter()
        if char then
            for _, tool in ipairs(char:GetChildren()) do
                if tool:IsA("Tool") then tool:Activate() end
            end
        end
    end)
end

local function ExpandHitboxes()
    for _, p in ipairs(Players:GetPlayers()) do
        if p ~= LocalPlayer and p.Character then
            local hrp = p.Character:FindFirstChild("HumanoidRootPart")
            if hrp then
                if not OrigHitboxes[p] then OrigHitboxes[p] = hrp.Size end
                pcall(function()
                    hrp.Size = Vector3.new(Config.HitboxSize, Config.HitboxSize, Config.HitboxSize)
                    hrp.Transparency = 0.7
                    hrp.CanCollide = false
                    hrp.Material = Enum.Material.ForceField
                end)
            end
        end
    end
end

ResetHitboxes = function()
    for p, size in pairs(OrigHitboxes) do
        pcall(function()
            if p.Character then
                local hrp = p.Character:FindFirstChild("HumanoidRootPart")
                if hrp then
                    hrp.Size = size
                    hrp.Transparency = 1
                    hrp.Material = Enum.Material.SmoothPlastic
                end
            end
        end)
    end
    OrigHitboxes = {}
end

local function RunAutoEquip()

                    end
                end
            end
        end
    end)
end

local function StartFly()
    local hrp = GetRootPart()
    local hum = GetHumanoid()
    if not hrp or not hum then return end
    FlyActive = true
    hum.PlatformStand = true
    FlyBV = Instance.new("BodyVelocity")
    FlyBV.MaxForce = Vector3.new(math.huge, math.huge, math.huge)
    FlyBV.Velocity = Vector3.new(0, 0, 0)
    FlyBV.Parent = hrp
    FlyBG = Instance.new("BodyGyro")
    FlyBG.MaxTorque = Vector3.new(math.huge, math.huge, math.huge)
    FlyBG.P = 9e4
    FlyBG.Parent = hrp
    Connections.FlyLoop = RunService.Heartbeat:Connect(function()
        if not FlyActive or not hrp or not hrp.Parent then StopFly() return end
        local dir = Vector3.new(0, 0, 0)
        if UserInputService:IsKeyDown(Enum.KeyCode.W) then dir = dir + Camera.CFrame.LookVector end
        if UserInputService:IsKeyDown(Enum.KeyCode.S) then dir = dir - Camera.CFrame.LookVector end
        if UserInputService:IsKeyDown(Enum.KeyCode.A) then dir = dir - Camera.CFrame.RightVector end
        if UserInputService:IsKeyDown(Enum.KeyCode.D) then dir = dir + Camera.CFrame.RightVector end
        if UserInputService:IsKeyDown(Enum.KeyCode.Space) then dir = dir + Vector3.new(0, 1, 0) end
        if UserInputService:IsKeyDown(Enum.KeyCode.LeftShift) then dir = dir - Vector3.new(0, 1, 0) end
        FlyBV.Velocity = dir * Config.FlySpeed
        FlyBG.CFrame = Camera.CFrame
    end)
end

StopFly = function()
    FlyActive = false
    if FlyBV then FlyBV:Destroy() FlyBV = nil end
    if FlyBG then FlyBG:Destroy() FlyBG = nil end
    if Connections.FlyLoop then Connections.FlyLoop:Disconnect() Connections.FlyLoop = nil end
    pcall(function() local hum = GetHumanoid() if hum then hum.PlatformStand = false end end)
end

local function RunNoclip()
    local char = GetCharacter()
    if not char then return end
    for _, part in ipairs(char:GetDescendants()) do
        if part:IsA("BasePart") then part.CanCollide = false end
    end
end

local function SetupInfiniteJump()
    if Connections.InfJump then return end
    Connections.InfJump = UserInputService.JumpRequest:Connect(function()
        if Config.InfiniteJump and IsAlive() then
            pcall(function() GetHumanoid():ChangeState(Enum.HumanoidStateType.Jumping) end)

end

local function RunAntiFling()
    local hrp = GetRootPart()
    if hrp and hrp.AssemblyLinearVelocity.Magnitude > 100 then
        hrp.AssemblyLinearVelocity = Vector3.new(0, 0, 0)
        hrp.AssemblyAngularVelocity = Vector3.new(0, 0, 0)
    end
end

local function RunAntiSlow()
    local hum = GetHumanoid()
    if not hum then return end
    if hum.WalkSpeed < 16 and not Config.SpeedEnabled then
        pcall(function() hum.WalkSpeed = 16 end)
    end
end

local function CreateESP(player)
    if player == LocalPlayer or ESPCache[player] then return end
    local hl = Instance.new("Highlight")
    hl.Name = "DrkBaediESP"
    hl.FillTransparency = 0.6
    hl.OutlineTransparency = 0
    hl.OutlineColor = Color3.fromRGB(200, 50, 50)
    hl.FillColor = Color3.fromRGB(150, 30, 30)

    local bb = Instance.new("BillboardGui")
    bb.Name = "DrkInfo"
    bb.Size = UDim2.fromOffset(200, 50)
    bb.StudsOffset = Vector3.new(0, 3.5, 0)
    bb.AlwaysOnTop = true
    bb.LightInfluence = 0

    local nl = Instance.new("TextLabel")
    nl.Size = UDim2.new(1, 0, 0.5, 0)
    nl.BackgroundTransparency = 1
    nl.TextColor3 = Color3.fromRGB(255, 255, 255)
    nl.TextStrokeTransparency = 0.3
    nl.TextStrokeColor3 = Color3.fromRGB(0, 0, 0)
    nl.Font = Enum.Font.GothamBold
    nl.TextSize = 13
    nl.Text = player.DisplayName
    nl.Parent = bb

    local il = Instance.new("TextLabel")
    il.Size = UDim2.new(1, 0, 0.5, 0)
    il.Position = UDim2.new(0, 0, 0.5, 0)
    il.BackgroundTransparency = 1
    il.TextColor3 = Color3.fromRGB(0, 255, 100)
    il.TextStrokeTransparency = 0.3
    il.TextStrokeColor3 = Color3.fromRGB(0, 0, 0)
    il.Font = Enum.Font.GothamBold
    il.TextSize = 11
    il.Text = "..."
    il.Parent = bb

    ESPCache[player] = { Highlight = hl, Billboard = bb, NameLabel = nl, InfoLabel = il }

    local function Attach()
        pcall(function()
            if player.Character then
                hl.Adornee = player.Character
                hl.Parent = player.Character
                local head = player.Character:FindFirstChild("Head")
                if head then bb.Adornee = head bb.Parent = head end
            end
        end)
    end
    Attach()
    Connections["ESP_" .. player.UserId] = player.CharacterAdded:Connect(function()
        task.wait(1)
        if Config.ESPEnabled then Attach() end
    end)
end

local function RemoveESP(player)
    local c = ESPCache[player]
    if c then
        pcall(function() c.Highlight:Destroy() end)
        pcall(function() c.Billboard:Destroy() end)
        ESPCache[player] = nil
    end
    local k = "ESP_" .. player.UserId
    if Connections[k] then Connections[k]:Disconnect() Connections[k] = nil end
end

EnableAllESP = function()
    for _, p in ipairs(Players:GetPlayers()) do CreateESP(p) end
    Connections.ESPJoin = Players.PlayerAdded:Connect(function(p)
        if Config.ESPEnabled then CreateESP(p) end
    end)
    Connections.ESPLeave = Players.PlayerRemoving:Connect(function(p) RemoveESP(p) end)
end

DisableAllESP = function()
    for p in pairs(ESPCache) do RemoveESP(p) end
    if Connections.ESPJoin then Connections.ESPJoin:Disconnect() Connections.ESPJoin = nil end
    if Connections.ESPLeave then Connections.ESPLeave:Disconnect() Connections.ESPLeave = nil end
end

UpdateESPInfo = function()
    local myRoot = GetRootPart()
    for player, data in pairs(ESPCache) do
        pcall(function()
            if player.Character then
                local hum = player.Character:FindFirstChildOfClass("Humanoid")
                if hum then
                    local pct = hum.Health / hum.MaxHealth
                    local col
                    if pct > 0.6 then col = Color3.fromRGB(0, 255, 80)
                    elseif pct > 0.3 then col = Color3.fromRGB(255, 200, 0)
                    else col = Color3.fromRGB(255, 50, 50) end
                    local distTxt = ""
                    if myRoot then
                        local er = player.Character:FindFirstChild("HumanoidRootPart")
                        if er then distTxt = " | " .. math.floor((myRoot.Position - er.Position).Magnitude) .. "m" end
                    end
                    data.InfoLabel.Text = string.format("HP: %d/%d%s", math.floor(hum.Health), math.floor(hum.MaxHealth), distTxt)
                    data.InfoLabel.TextColor3 = col
                    data.Highlight.FillColor = col
                end
            end
        end)
    end
end

local function DisableBallESP()
    if BallHighlight then BallHighlight:Destroy() BallHighlight = nil end
    if BallBillboard then BallBillboard:Destroy() BallBillboard = nil end
end

local function UpdateBallESP()
    if not Config.BallESP then return end
    local ball = FindBall()
    if ball and BallHighlight then
        pcall(function()
            BallHighlight.Adornee = ball
            BallHighlight.Parent = ball
            BallBillboard.Adornee = ball
            BallBillboard.Parent = ball

            local myRoot = GetRootPart()
            if myRoot then
                local dist = math.floor((ball.Position - myRoot.Position).Magnitude)
                local speed = math.floor(GetBallSpeed(ball))
                local targeting = IsBallComingToMe(ball)
                local label = BallBillboard:FindFirstChildOfClass("TextLabel")
                if label then
                    label.Text = "BALL | " .. dist .. "m | " .. speed .. " spd"
                    if targeting then
                        label.TextColor3 = Color3.fromRGB(255, 50, 50)
                        BallHighlight.FillColor = Color3.fromRGB(255, 0, 0)
                    else
                        label.TextColor3 = Color3.fromRGB(255, 255, 0)
                        BallHighlight.FillColor = Color3.fromRGB(255, 200, 0)
                    end
                end
            end
        end)
    end
end

EnableFullbright = function()
    pcall(function()
        Lighting.Ambient = Color3.fromRGB(200, 200, 200)
        Lighting.Brightness = 2
        Lighting.FogEnd = 1e6
        Lighting.GlobalShadows = false
        for _, v in ipairs(Lighting:GetDescendants()) do
            if v:IsA("BloomEffect") or v:IsA("BlurEffect") or v:IsA("ColorCorrectionEffect") or v:IsA("DepthOfFieldEffect") or v:IsA("SunRaysEffect") then
                v.Enabled = false
            end
        end
    end)
end

DisableFullbright = function()
    pcall(function()
        Lighting.Ambient = OriginalAmbient
        Lighting.Brightness = OriginalBrightness
        Lighting.GlobalShadows = true
        for _, v in ipairs(Lighting:GetDescendants()) do
            if v:IsA("BloomEffect") or v:IsA("BlurEffect") or v:IsA("ColorCorrectionEffect") or v:IsA("DepthOfFieldEffect") or v:IsA("SunRaysEffect") then
                v.Enabled = true
            end
        end
    end)
end

Window:AddButton({ Title = "Teleport to Player", Description = "TP to named player", Tab = PlayerTab,
    Callback = function()
        if Config.TeleportTarget == "" then Window:Notify({ Title = "Error", Description = "Enter a name!", Duration = 3 }) return end
        local myRoot = GetRootPart() if not myRoot then return end
        local target = FindPlayerByName(Config.TeleportTarget)
        if target and target.Character then
            local tRoot = target.Character:FindFirstChild("HumanoidRootPart")
            if tRoot then myRoot.CFrame = tRoot.CFrame * CFrame.new(0, 0, 3)
                Window:Notify({ Title = "Done", Description = "TP to " .. target.DisplayName, Duration = 3 }) return end
        end
        Window:Notify({ Title = "Not Found", Description = "Player not found", Duration = 3 })
    end })

Window:AddButton({ Title = "Teleport to Nearest", Description = "TP behind closest player", Tab = PlayerTab,
    Callback = function()

            local tRoot = target.Character:FindFirstChild("HumanoidRootPart")

local espClock = 0
local hitboxClock = 0
local ballEspClock = 0
local equipClock = 0

Connections.MainLoop = RunService.Heartbeat:Connect(function(dt)
    if not GetCharacter() or not IsAlive() then return end

    if Config.SmartParry then pcall(RunSmartParry)
    elseif Config.AutoParry then pcall(RunAutoParry) end

    if Config.AutoDodge then pcall(RunAutoDodge) end
    if Config.AutoSpam then pcall(RunAutoSpam) end

    hitboxClock = hitboxClock + dt
    if Config.HitboxExpander and hitboxClock >= 0.5 then hitboxClock = 0 pcall(ExpandHitboxes) end

    equipClock = equipClock + dt
    if Config.AutoEquip and equipClock >= 1 then equipClock = 0 pcall(RunAutoEquip) end

    if Config.SpeedEnabled then pcall(function() GetHumanoid().WalkSpeed = Config.SpeedValue end) end
    if Config.JumpEnabled then pcall(function() local h = GetHumanoid() h.UseJumpPower = true h.JumpPower = Config.JumpValue end) end
    if Config.NoclipEnabled then pcall(RunNoclip) end
    if Config.AntiFling then pcall(RunAntiFling) end
    if Config.AntiSlow then pcall(RunAntiSlow) end
    if Config.FOVEnabled then pcall(function() Camera.FieldOfView = Config.FOVValue end) end

    espClock = espClock + dt
    if Config.ESPEnabled and espClock >= 0.25 then espClock = 0 pcall(UpdateESPInfo) end

    ballEspClock = ballEspClock + dt
    if Config.BallESP and ballEspClock >= 0.1 then ballEspClock = 0 pcall(UpdateBallESP) end
end)

Connections.Respawn = LocalPlayer.CharacterAdded:Connect(function(char)
    task.wait(1.5)
    if Config.SpeedEnabled then pcall(function() char:FindFirstChildOfClass("Humanoid").WalkSpeed = Config.SpeedValue end) end
    if Config.JumpEnabled then pcall(function() local h = char:FindFirstChildOfClass("Humanoid") h.UseJumpPower = true h.JumpPower = Config.JumpValue end) end
    if Config.FlyEnabled then StopFly() task.wait(0.5) StartFly() end
    if Config.AutoEquip then task.delay(1, function() pcall(RunAutoEquip) end) end
    task.delay(2, function() pcall(FindParryRemote) end)
end)

        btn.Size = UDim2.fromOffset(52, 52)
        btn.Position = UDim2.new(0, 14, 0.4, 0)
        btn.AnchorPoint = Vector2.new(0, 0.5)
        btn.BackgroundColor3 = Color3.fromRGB(35, 12, 12)
        btn.TextColor3 = Color3.fromRGB(255, 100, 100)
        btn.Text = "D"
        btn.Font = Enum.Font.GothamBlack
        btn.TextSize = 26
        btn.AutoButtonColor = true
        btn.Parent = gui
        Instance.new("UICorner", btn).CornerRadius = UDim.new(0, 14)
        local st = Instance.new("UIStroke", btn)
        st.Color = Color3.fromRGB(200, 50, 50)
        st.Thickness = 2

        local dragging, dragStart, startPos, wasDragged = false, nil, nil, false
        btn.InputBegan:Connect(function(input)
            if input.UserInputType == Enum.UserInputType.Touch or input.UserInputType == Enum.UserInputType.MouseButton1 then
                dragging = true wasDragged = false dragStart = input.Position startPos = btn.Position
                input.Changed:Connect(function() if input.UserInputState == Enum.UserInputState.End then dragging = false end end)
            end
        end)
        btn.InputChanged:Connect(function(input)
            if dragging and (input.UserInputType == Enum.UserInputType.Touch or input.UserInputType == Enum.UserInputType.MouseMovement) then
                local delta = input.Position - dragStart
                if delta.Magnitude > 8 then wasDragged = true
                    btn.Position = UDim2.new(startPos.X.Scale, startPos.X.Offset + delta.X, startPos.Y.Scale, startPos.Y.Offset + delta.Y) end
            end
        end)
        btn.MouseButton1Click:Connect(function()
            if not wasDragged then SimulateKey(Enum.KeyCode.LeftAlt, 0.05) end
        end)
    end)
end
