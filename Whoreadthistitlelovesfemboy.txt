local function fireNativeKnit(sn, mn)
    pcall(function()
        local km = ReplicatedStorage:FindFirstChild("Packages") and ReplicatedStorage.Packages:FindFirstChild("knit") or ReplicatedStorage:FindFirstChild("Knit")
        if km then
            local k = require(km); local s = k.GetService(sn)
            if s and s[mn] then
                if typeof(s[mn]) == "table" and s[mn].Fire then s[mn]:Fire()
                elseif type(s[mn]) == "function" then s[mn](s) end
            end
        end
    end)
end

local function cutGrassOnce()
    local z = workspace:FindFirstChild("Zones"); if not z then return end
    for _, zone in ipairs(z:GetChildren()) do
        if string.find(zone.Name, "Lava_") then
            for _, obj in ipairs(zone:GetDescendants()) do
                if string.find(obj.Name, "Grass_") then obj:Destroy() end
            end
        end
    end
end

local function scanAndTakeRarity(targetRarity, preloadCFrame)
    local hrp = player.Character and player.Character:FindFirstChild("HumanoidRootPart")
    if not hrp then return end
    local myBase = workspace.Bases:FindFirstChild(player.Name)
    if preloadCFrame then hrp.CFrame = preloadCFrame + Vector3.new(0,5,0); hrp.Anchored = true; task.wait(0.2) end
    local zones = workspace:FindFirstChild("Zones")
    if not zones then hrp.Anchored = false return end
    local acquired = false
    while states["Auto"..targetRarity] and not acquired do
        for _, zone in ipairs(zones:GetChildren()) do
            if string.find(zone.Name, "Lava_") then
                local sp = zone:FindFirstChild("SpawnZone")
                if sp then
                    for _, item in ipairs(sp:GetChildren()) do
                        local match = false
                        pcall(function() if string.find(string.upper(item.BrainrotInfo.Info.Rarity.Text), string.upper(targetRarity)) then match = true end end)
                        if match then
                            local tp = item:FindFirstChild("Take", true)
                            if tp and tp:IsA("ProximityPrompt") then
                                hrp.Anchored = false; hrp.CFrame = item:GetPivot(); task.wait(0.1)
                                fireproximityprompt(tp); task.wait(0.2); acquired = true; break
                            end
                        end
                    end
                end
            end
            if acquired then break end
        end
        if not acquired then task.wait(0.5) end
    end
    hrp.Anchored = false
    if acquired and myBase then
        hrp.CFrame = myBase:GetPivot() * CFrame.new(0,3,0); task.wait(0.2)
        pcall(function() local bt = workspace.Zones.Base_Zone.Base_Zone; firetouchinterest(hrp, bt, 0); firetouchinterest(hrp, bt, 1) end)
    end
end

-- ============ DRAGGING ============
local dragging, dragInput, dragStart, startPos

Top.InputBegan:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
        dragging = true
        dragStart = input.Position
        startPos = Main.AbsolutePosition 
    end
end)

Top.InputChanged:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch then
        dragInput = input
    end
end)

UIS.InputChanged:Connect(function(input)
    if input == dragInput and dragging then
        local d = input.Position - dragStart
        local nx = math.clamp(startPos.X + d.X, 0, scrW - W)
        local ny = math.clamp(startPos.Y + d.Y, 0, scrH - topH)
        Main.Position = UDim2.new(0, nx, 0, ny)
    end
end)

UIS.InputEnded:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
        dragging = false
    end
end)

    if fD and i.UserInputType == Enum.UserInputType.Touch then

        FB.Position = UDim2.new(0, math.clamp(fSP.X.Offset + d.X, 0, scrW - 50), 0, math.clamp(fSP.Y.Offset + d.Y, 0, scrH - 50))
    end

    if i.UserInputType == Enum.UserInputType.Touch and fD then

    end
